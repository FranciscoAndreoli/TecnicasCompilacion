int suma(int x, int y) {
    return x + y;
}

int main() {
    int a;
    int b;
    int c;
    a = 456;
    b = 34 + 65 - 18;
    if (b > a) {
        c = b;
    } else {
        c = a;
    }
    for (int i = 0; i < 10; i = i + 1) {
        c = c + i;
    }
    while (a < c) {
        a = a + 1;
    }
    c = suma(a, b);
    return c;
}
